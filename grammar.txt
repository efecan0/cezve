parse: (KEYWORD[0] | KEYWORD[1] | KEYWORD[2]) assign
      : expr

expr  : term ((PLUS | MINUS) term)*

term  : factor  ((MUL|DIV) factor)*

factor : INT|FLOAT
        : (PLUS|MINUS) factor
        : LPAREN expr RPAREN

assign : key

key   : int IDENT ASSIGN INT
        : float IDENT ASSIGN FLOAT
        : string IDENT ASSIGN string
